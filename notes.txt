//moves box across the screen until 100px
function pointGrab() {
  var pointGrabber = document.getElementById("point-grabber");
  var position = 0;
  var id = setInterval(function(){
    if (position == 1000) {
      clearInterval(id);
    } else {
      position++;
      pointGrabber.style.left = position + 'px';
      // pointGrabber.style.left = position + 'px';
      // pointGrabber.style.right = position + 'px';
    }
  }, 5);
}
